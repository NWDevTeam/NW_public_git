	add_equipment_to_stockpile = {
		type = infantry_equipment
		amount = 10000
	}
	if = {
		limit = { NOT = { has_template = "Militia Brigada" } }

			name = "Militia Brigada"
			is_locked = no
			regiments = {
				infantry = { x = 0 y = 0 }
				infantry = { x = 0 y = 1 }
				infantry = { x = 0 y = 2 }
				infantry = { x = 0 y = 3 }
			}
		}
	}
	every_state = {
		limit = {
			OR = {
				is_core_of = SPR
				is_core_of = SPC
				is_core_of = SEU
				is_core_of = NAV
				is_core_of = ARA
				is_core_of = ANL
				is_core_of = CAS
				is_core_of = ASU
				is_core_of = SNT
				is_core_of = CAT
				is_core_of = BAS
				is_core_of = GLC
				is_core_of = VLA
			}
		}
		add_core_of = PREV
	}
	every_state = {
		limit = {
			OR = {
				is_owned_by = SPR
				is_owned_by = SPC
				is_owned_by = SEU
				is_owned_by = NAV
				is_owned_by = ARA
				is_owned_by = ANL
				is_owned_by = CAS
				is_owned_by = ASU
				is_owned_by = SNT
				is_owned_by = CAT
				is_owned_by = BAS
				is_owned_by = GLC
				is_owned_by = VLA
			}
		}
		add_claim_by = PREV
	}
	inherit_technology = SPR
	set_stability = 0.9
	set_war_support = 0.5
	log = "[GetDateText]: [Root.GetName]: give_rifles_and_template"
}
