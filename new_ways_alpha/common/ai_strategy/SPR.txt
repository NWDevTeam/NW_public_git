# make italy and germany want to send stuff to the fascist side of SCW
ITA_and_GER_help_SPA_win_civil_war = {
	allowed = {
		OR = {
			original_tag = GER
			original_tag = ITA
			original_tag = FRA
			original_tag = JAP
		}
	}
	enable = {
		date < 1939.1.1
		country_exists = SPR
		SPR = {
			has_government = fascism
		}
		has_government = fascism
		SPR = { has_war_with = SPA }
		NOT = { has_global_flag = scw_over }
	}
	abort = {
		OR = {
			date > 1939.1.1
			NOT = { has_government = fascism }
			NOT = { SPR = { has_war_with = SPA } }
			has_global_flag = scw_over
		}
	}
	
	ai_strategy = {
		type = send_volunteers_desire
		id = SPR
		value = -50
	}

}

# Prevent Nationalist Spain to join any faction on historical mode
nationalist_spain_allies_nobody_on_historical = {
	allowed = {
		original_tag = SPA
	}
	enable = {
		has_global_flag = scw_over
		OR = {
			AND = {
				is_historical_focus_on = yes
				OR = {
					has_game_rule = {
						rule = SPR_ai_behavior 
						option = DEFAULT
					}
					has_game_rule = {
						rule = SPR_ai_behavior 
						option = COMMUNIST
					}
				}
			}
			SPR = { has_country_flag = NW_SPR_AI_COMMUNIST }
			has_game_rule = {
				rule = SPA_ai_behavior 
				option = DEFAULT
			}
			has_game_rule = {
				rule = SPA_ai_behavior 
				option = FASCIST
			}
			has_country_flag = NW_SPA_AI_FASCIST
			has_game_rule = {
				rule = SPA_ai_behavior 
				option = FASCIST_ALTERNATE
			}
			has_country_flag = NW_SPA_AI_FASCIST_ALTERNATE
		}
	}
	abort = {
		OR = {
			SPA = { has_war = yes }
			SPA = { is_in_faction = yes }
			GER = { NOT = { has_government = fascism } }
			ENG = { NOT = { has_government = democratic } }
		}
	}
	
	ai_strategy = {
		type = alliance
		id = GER
		value = -9999
	}

	ai_strategy = {
		type = alliance
		id = ENG
		value = -9999
	}
}
