SWE_want_greatness_FIN = {
	allowed = {
		original_tag = SWE
		is_historical_focus_on = no
	}
	enable = {
		is_subject = no
		has_civil_war = no
		has_war = no
		NOT = {
			has_war_with = NOR
			has_war_with = DEN
		}
		OR = {
			num_divisions > 50
			strength_ratio = { tag = FIN ratio > 1 }
			FIN = { has_civil_war = yes }
		}
		FIN = {
			is_subject = no
			is_in_faction = no
			is_neighbor_of = ROOT
		}
		OR = {
			NOT = { has_government = democratic }
			FIN = { NOT = { has_government = democratic } }
		}
	}
	abort_when_not_enabled = yes
	
	ai_strategy = {
		type = antagonize
		id = "FIN"
		value = 200
	}
	ai_strategy = {
		type = conquer
		id = "FIN"
		value = 200
	}
	ai_strategy = {
		type = declare_war
		id = "FIN"
		value = 200
	}
}
