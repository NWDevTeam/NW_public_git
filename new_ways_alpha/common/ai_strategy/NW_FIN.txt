			has_government = fascism

			has_government = neutrality
			has_government = democratic
		}
		OR = {
			AND = {
				has_war_together_with = GER
				has_war_with = SOV
			}
			AND = {
				has_country_flag = NW_fin_continuation_war_flag

}
