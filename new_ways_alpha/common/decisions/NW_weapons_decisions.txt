	sell_infantry_weapons = {

		icon = generic_prepare_civil_war

		available = {
			has_equipment = { infantry_equipment > 500 }
		}
		
		days_re_enable = 30
		#fire_only_once = yes
	
		custom_cost_trigger = {
			set_temp_variable = { base_weapons_sell_cost = 15 }
			set_temp_variable = { threat_for_cost = global.threat }
			multiply_temp_variable = { threat_for_cost = 25 }
			add_to_temp_variable = { base_weapons_sell_cost = threat_for_cost }
			if = {	#скидка
				limit = { check_variable = { var = weapons_sell_discount value = 0 compare = not_equals } }
				multiply_temp_variable = { base_weapons_sell_cost = weapons_sell_discount }
			}
		}
		custom_cost_text = sell_weapons_cost
		
		#cost = 50
		
		complete_effect = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -500
			}
			set_temp_variable = { base_weapons_sell_cost = 25 }
			set_temp_variable = { threat_for_cost = global.threat }
			multiply_temp_variable = { threat_for_cost = 50 }
			add_to_temp_variable = { base_weapons_sell_cost = threat_for_cost }
			if = {	#скидка
				limit = { check_variable = { var = weapons_sell_discount value = 0 compare = not_equals } }
				multiply_temp_variable = { base_weapons_sell_cost = weapons_sell_discount }
			}
			round_variable = base_weapons_sell_cost
			add_political_power = base_weapons_sell_cost
			
			hidden_effect = {
				add_to_variable = {
					var = global.total_infantry_weapons
					value = 500
				}
			}
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): complete decision sell_infantry_weapons // total_infantry_weapons = [?global.total_infantry_weapons]"
		}
		
		ai_will_do = {
			factor = 0
			modifier = {
				add = 10
				stockpile_ratio = {
					archetype = infantry_equipment
					ratio > 1
				}
			}
			modifier = {
				add = 50
				has_equipment = { infantry_equipment > 10000 }
			}
			modifier = {
				add = 1
				has_war = no
			}
			modifier = {
				add = 15
				has_political_power < 0
				stockpile_ratio = {
					archetype = infantry_equipment
					ratio > 0.5
				}
			}
			modifier = {
				factor = 0

			}
			modifier = {
				factor = 0
				has_equipment = { infantry_equipment < 1000 }
