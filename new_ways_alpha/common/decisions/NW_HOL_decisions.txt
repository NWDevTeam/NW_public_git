		custom_cost_text = custom_cost_anarcho_unity
		fire_only_once = no
		available = {

			}
		}
		target_array = global.countries
		targets_dynamic = yes
		complete_effect = {
			log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): complete decision HOL_preemptive_strike_decision"

			if = {
				limit = { NOT = { has_country_flag = HOL_preemptive_strike_decision_flag_for_achievement } }

			}
			declare_war_on = { target = FROM type = annex_everything }
		}
	}
