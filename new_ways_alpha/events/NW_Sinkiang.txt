country_event = {
	id = NWsinkiang.2
	title = NWsinkiang.2.t
	desc = NWsinkiang.2.desc
	picture = GFX_report_event_chinese_soldiers
	fire_only_once = yes
	
	trigger = {
		tag = SIK
	    if = {
		    limit = {
		     	AND = {
    				has_country_flag = SIK_muhiti_increased_influence_1
			        has_country_flag = SIK_muhiti_bought_rifles
                }		    
			}
		    date > 1936.10.05
		}
        else_if = {
            limit = {
                OR = {
				    has_country_flag = SIK_muhiti_increased_influence_1
			        has_country_flag = SIK_muhiti_bought_rifles
	            }
	        }
			date > 1937.01.05
		}
        else = {
 		    date > 1937.04.05
		}
    }		
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWsinkiang.2" }
	immediate = {
	    UYG = {
			transfer_state = 619
		    inherit_technology = SIK
		    load_oob = "UYG_1936"
		}
	    every_unit_leader = {
		    limit = {
			    OR = {
				    has_id = 15555
			        has_id = 15556
					has_id = 15557
		        }
			}
		    set_nationality = UYG
	    }
		if = {
		    limit = {
		     	AND = {
    				has_country_flag = SIK_muhiti_increased_influence_1
			        has_country_flag = SIK_muhiti_bought_rifles
                }		    
			}
		    capital_scope = {
			    create_unit = {
			    	division = "name = \"Uyghur Militia\" division_template = \"Juntuán\" start_experience_factor = 0.4 start_equipment_factor = 0.8"
			        owner = UYG
					count = 2
					allow_spawning_on_enemy_provs = yes 
			    }
			}	
		}
        else_if = {
            limit = {
                OR = {
				    has_country_flag = SIK_muhiti_increased_influence_1
			        has_country_flag = SIK_muhiti_bought_rifles
	            }
	        }
			capital_scope = {
			    create_unit = {
			    	division = "name = \"Uyghur Militia\" division_template = \"Juntuán\" start_experience_factor = 0.4 start_equipment_factor = 0.8"
			        owner = UYG
					allow_spawning_on_enemy_provs = yes
			    }
			}	
		}
		declare_war_on = {
			target = UYG
			type = civil_war
		}
		add_civil_war_target = UYG
		hidden_effect = {
			617 = {
				create_unit = {
					division = "name = \"Kirgiz Brigade\" division_template = \"Juntuán\" start_experience_factor = 0.45" 
					owner = ROOT
					count = 3
					allow_spawning_on_enemy_provs = yes 
				}
			}
		}
	}
	option = {
		name = NWsinkiang.2.a
	}	
	option = {
		name = NWsinkiang.2.b
		trigger = { is_ai = no }
        UYG = { change_tag_from = ROOT }		
    }
}
