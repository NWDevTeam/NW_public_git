	#Ok - Already completed initial missiolini focus
	option = {
		name = BBA_italy_second_italo_ethiopian_war.2.a
		hidden_effect = {
			every_controlled_state = {
				limit = {
					is_core_of = FROM
				}
				set_state_controller_to = FROM
				FROM = { transfer_state = PREV }
			}
		}
		FROM = {
			#ERITREA
			transfer_state = 550
			550 = { add_compliance = 50 }
			
			#SOMALILAND
			transfer_state = 559
			559 = { add_compliance = 50 }
			
			#TIGRAY
			transfer_state = 1080
			
			#AFAR always goes to ETH if owned by ITA
			if = {
				limit = {
					ROOT = { owns_state = 1079 }
				}
				transfer_state = 1079 
			}
			white_peace = ROOT
		}
		hidden_effect = {
			FROM = {
				every_enemy_country = {
					limit = {
						is_in_faction_with = ROOT
						is_faction_leader = yes
					}
					FROM = { white_peace = PREV }
				}
			}
			if = {
				limit = { NOT = { country_exists = ETH } }
				every_controlled_state = {
					limit = {
						is_core_of = FROM
					}
					FROM = { transfer_state = PREV }
				}
				FROM = {
					#ERITREA
					transfer_state = 550
					550 = { add_compliance = 50 }
					
					#SOMALILAND
					transfer_state = 559
					559 = { add_compliance = 50 }
					
					#TIGRAY
					transfer_state = 1080
					
					#AFAR always goes to ETH if owned by ITA
					if = {
						limit = {
							ROOT = { owns_state = 1079 }
						}
						transfer_state = 1079 
					}
				}
			}
		}

		ITA_bop_low_decrease_effect = yes
	}
}
