	}

	immediate = {
		hidden_effect = {
			set_global_flag = FAW_flag
			random_country = {
				limit = {
					has_government = anarchism
					has_defensive_war = yes
					any_enemy_country = {
						has_government = anarchism
						has_offensive_war_with = PREV
					}

					limit = {
						has_government = anarchism
						has_offensive_war_with = PREV
					}
					save_event_target_as = is_attacker_from
				}
			}
		}
	}
	
	option = {
		name = trworldnews.136.a
		trigger = {
			OR = {
				tag = FROM
				tag = event_target:is_attacker_from
			}

		name = trworldnews.136.b
