	id = NWmonarchy.2
	title = NWmonarchy.2.t
	desc = NWmonarchy.2.desc
	is_triggered_only = yes
	
	immediate = { log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWmonarchy.2" }

	option = {
		log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): choose option NWmonarchy.2.a"
		name = NWmonarchy.2.a
		ai_chance = {
			factor = 1
		}
		if = {
			limit = { monarchy > 0.499 }
			every_state = {
				limit = {
					is_core_of = ROOT
				}
				add_core_of = FROM
			}
			ROOT = {
				every_unit_leader = {
					set_nationality = FROM
				}
			}
			if = {
				limit = { is_ai = no }
				FROM = { change_tag_from = ROOT }
			}
			FROM = {
				annex_country = {
					target = ROOT
					transfer_troops = yes
				}
			}
			if = {
				limit = { original_tag = CZE }
				FROM = {
					set_country_flag = CZE_transfer_ideas
					set_country_flag = kingCZE_flag
					add_country_leader_trait = kingCZE
				}
			}
			else_if = {
				limit = { original_tag = AUS }
				FROM = {
					set_country_flag = AUS_transfer_ideas
					set_country_flag = kingAUS_flag
					add_country_leader_trait = kingAUS
				}
			}
			else_if = {
				limit = { original_tag = HUN }
				FROM = {
					set_country_flag = HUN_transfer_ideas
					set_country_flag = kingHUN_flag
					add_country_leader_trait = kingHUN
				}
			}
		}
		else_if = {
			limit = {
				AND = {
					monarchy > 0.299
					monarchy < 0.500
				}
			}
			if = {
				limit = {
					OR = {
						original_tag = CZE
						original_tag = AUS
						original_tag = HUN
					}
				}
				set_country_flag = hbbrg_minor_flag
			}
			if = {
				limit = { FROM = { has_government = fascism } }
				FROM = {
					set_autonomy = {
						target = ROOT
						autonomy_state = autonomy_reichsprotectorate
					}
				}
				else_if = {
					limit = { FROM = { has_government = monarchy } }
					FROM = {
						set_autonomy = {
							target = ROOT
							autonomy_state = autonomy_habs_imperial_uniate
						}
					}
				}
				else_if = {
					limit = { FROM = { NOT = { has_government = fascism has_government = monarchy } } }
					FROM = {
						set_autonomy = {
							target = ROOT
							autonomy_state = autonomy_puppet
						}
					}
				}
			}
		}
		else_if = {
			limit = { monarchy < 0.300 }
			FROM = { country_event = { id = NWmonarchy.3 days = 1 } }
		}
	}
	
	option = {
		log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): choose option NWmonarchy.1.b"
		name = NWmonarchy.2.b
		ai_chance = {
			factor = 99
		}
		trigger = {
			tag = HUN
			has_completed_focus = HUN_proclaim_greater_hungary
		}
		FROM = { country_event = { id = NWmonarchy.3 days = 1 } }
	}
}

	}

		NOT = { has_focus_tree = hbbrg_focus }
		has_government = monarchy
		NOT = { has_country_flag = { flag = HBBRG_solution_on_root value > 0 } }
		OR = {
			AND = {
				capital_scope = { is_on_continent = europe }
				NOT = { is_in_array = { global.HBBRG_monarchy_of_europe_array = ROOT } }
			}
			AND = {
				NOT = { capital_scope = { is_on_continent = europe } }
				is_in_array = { global.HBBRG_monarchy_of_europe_array = ROOT }
			}
		}
	}
	
	immediate = {
		log = "[GetDateText]: [Root.GetName] ([Root.GetTag]): event NWmonarchy.103"

			limit = {
				capital_scope = { is_on_continent = europe }
				NOT = { is_in_array = { global.HBBRG_monarchy_of_europe_array = ROOT } }
			}
			add_to_array = { global.HBBRG_monarchy_of_europe_array = ROOT }
		}
		else_if = {
			limit = {
				NOT = { capital_scope = { is_on_continent = europe } }

			}

	}
